SET SERVEROUTPUT ON;

DECLARE
  -- Input values
  v_subject_opted Student.subject_opted%TYPE := 'DBMS';
  v_subject_code  Student.subject_code%TYPE  := 'S01';

  -- Cursor declaration
  CURSOR stu_cur IS
    SELECT roll_no, name, subject_opted, subject_code
    FROM Student
    WHERE subject_opted = v_subject_opted
      AND subject_code  = v_subject_code;

  v_row stu_cur%ROWTYPE;
BEGIN
  OPEN stu_cur;
  LOOP
    FETCH stu_cur INTO v_row;
    EXIT WHEN stu_cur%NOTFOUND;

    DBMS_OUTPUT.PUT_LINE(
      'Roll No: ' || v_row.roll_no ||
      ', Name: ' || v_row.name ||
      ', Subject: ' || v_row.subject_opted ||
      ', Code: ' || v_row.subject_code
    );
  END LOOP;
  CLOSE stu_cur;
END;
